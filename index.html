function scrollToNext(){ const transition = document.getElementById("transition"); const invite = document.getElementById("invite"); const next = document.getElementById("nextPage"); // 🔥 ONLY NEW LUXURY ANIMATION HERE transition.classList.add("active"); setTimeout(()=>{ invite.classList.add("page-exit"); setTimeout(()=>{ invite.classList.remove("active","page-exit"); invite.style.display = "none"; next.style.display = "flex"; next.classList.add("active","page-enter"); setTimeout(()=>{ next.classList.remove("page-enter"); },1000); },300); setTimeout(()=>{ transition.classList.remove("active"); },900); },150); }
